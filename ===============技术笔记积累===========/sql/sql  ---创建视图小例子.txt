视图是一个虚拟的表，在前边加上CREATE OR REPLACE VIEW后边加上视图名即可

drop view V_DAILY_RECKONING_INFO

CREATE OR REPLACE VIEW v_daily_reckoning_info AS
SELECT T.DAILY_RECKONING_ID,
  T.USER_ID,
  T.USER_NAME,
   T.RECORD_DATE,
   T.RECKONING_DATE,
   O.USER_ID ORGANIZATION_ID,
   O.USER_NAME ORGANIZATION_NAME,
   T.USER_MOME,
   T.BIZ_TYPE,
   T.GW_TYPE,
  (SELECT DISPLAY_VALUE FROM  PB_LIST_OF_VALUE WHERE STAND_CODE=TO_CHAR(T.BIZ_TYPE) AND TYPE_ID='bizType') AS BIZTYPE_NAME,
  (SELECT DISPLAY_VALUE FROM  PB_LIST_OF_VALUE WHERE STAND_CODE=TO_CHAR(T.GW_TYPE) AND TYPE_ID='MOBILE_COM') AS MOBILE_COM_NAME,
  T.SEND_QUANTITY,
  T.UNIT_PRICE,
  T.SUB_TOTAL_FEE
 FROM DAILY_RECKONING_INFO T
 LEFT JOIN USER_INFO U ON T.USER_ID=U.USER_ID
 LEFT JOIN ORGANIZATION O ON U.ORGANIZATION_ID=O.USER_ID
;
